// Write a program to interface LEDs on pin no. 10,11,12,13 and blink alternatively at the
delay of 1 sec.
void setup() 
{
pinMode(10, OUTPUT);
pinMode(11, OUTPUT);
pinMode(12, OUTPUT);
pinMode(13, OUTPUT);
}
void loop() 
{
digitalWrite(10, HIGH);
delay(1000);
digitalWrite(10, LOW);

digitalWrite(11, HIGH);
delay(1000);
digitalWrite(11, LOW);

digitalWrite(12, HIGH);
delay(1000);
digitalWrite(12, LOW);

digitalWrite(13, HIGH);
delay(1000);
digitalWrite(13, LOW);
}
